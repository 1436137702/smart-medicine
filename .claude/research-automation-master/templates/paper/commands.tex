% ==============================================================================
% FIGURE INSERTION COMMANDS
% ==============================================================================

% Main figure
\newcommand{\insertfigure}[3]{%
  \end{multicols*}
  \setlength{\parindent}{0pt}
  \setlength{\intextsep}{0pt}
  \vspace{-5pt}
  \begin{minipage}{\linewidth}
  \begin{figure}[H]
  \includegraphics[width=0.999\linewidth]{#1}
  \setlength{\abovecaptionskip}{0pt}
  \justifying
  \myriadpro\fontsize{8pt}{9.6pt}\selectfont
  \caption{\textbf{#2} #3}
  \end{figure}
  \end{minipage}
  \vspace{-8pt}
  \setlength{\intextsep}{12pt}
  \setlength{\abovecaptionskip}{2pt}
  \begin{multicols*}{2}
}

% Supplementary figure
\newcommand{\insertsupfigure}[3]{%
  \end{multicols*}
  \setlength{\parindent}{0pt}
  \setlength{\intextsep}{0pt}
  \vspace{-5pt}
  \begin{minipage}{\linewidth}
  \renewcommand{\figurename}{Supplementary Figure}
  \begin{figure}[H]
  \includegraphics[width=0.999\linewidth]{#1}
  \setlength{\abovecaptionskip}{0pt}
  \justifying
  \myriadpro\fontsize{8pt}{9.6pt}\selectfont
  \caption{\textbf{#2} #3}
  \end{figure}
  \end{minipage}
  \vspace{-8pt}
  \setlength{\intextsep}{12pt}
  \setlength{\abovecaptionskip}{2pt}
  \begin{multicols*}{2}
}
