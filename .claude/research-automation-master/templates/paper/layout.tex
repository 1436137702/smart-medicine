% ==============================================================================
% PAGE GEOMETRY
% ==============================================================================
\geometry{
  a4paper,
  left=15mm,
  right=16mm,
  top=20mm,
  bottom=14mm,
  columnsep=5mm
}

\setlength{\headheight}{0mm}

% ==============================================================================
% COLORS
% ==============================================================================
\definecolor{mycolor}{rgb}{0.3,0.5,0.2}
\definecolor{linkblue}{RGB}{39,91,155}

% ==============================================================================
% AUTHOR/AFFILIATION FORMATTING
% ==============================================================================
\renewcommand\Affilfont{\raggedright\fontsize{7}{8}\selectfont\itshape}
\renewcommand\Authfont{\raggedright\normalsize}
\renewcommand\Authands{ and \vspace{-3mm}}

% ==============================================================================
% SECTION FORMATTING
% ==============================================================================
\titleformat{\section}
  {\normalfont\fontsize{12}{12}\bfseries\color{mycolor}}
  {}{0em}{}
\titlespacing*{\section}{0pt}{4pt}{0pt}

\titleformat{\subsection}
  {\normalfont\fontsize{11}{5}\bfseries\color{mycolor}}
  {}{0em}{\textit}
\titlespacing{\subsection}{0pt}{5pt}{0pt}

\titleformat{\subsubsection}
  {\normalfont\fontsize{11}{5}\bfseries\color{mycolor}}
  {}{0em}{}
\titlespacing{\subsubsection}{0pt}{5pt}{-2pt}

% ==============================================================================
% PARAGRAPH SPACING
% ==============================================================================
\setlength{\parskip}{3pt}

% ==============================================================================
% HEADER AND FOOTER
% ==============================================================================
\fancyhf{}
\fancyfoot[L]{\hspace{-1mm}\fontsize{8.5}{8.4}\selectfont\textbf{\firstauthorsurname\ et al.}}
\fancyfoot[C]{\hspace{21mm}\fontsize{8}{8.4}\selectfont\textit{\articletitle}}
\fancyfoot[R]{\small\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\pagestyle{fancy}

% First page style
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[L]{\hspace{-1mm}\fontsize{8.5}{8.4}\selectfont\textbf{\firstauthorsurname\ et al.}}
  \fancyfoot[C]{\hspace{21mm}\fontsize{8}{8.4}\selectfont\textit{\articletitle}}
  \fancyfoot[R]{\small\thepage}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

\setlength{\footskip}{-6mm}

% ==============================================================================
% PAGE DECORATIONS (colored lines)
% ==============================================================================

% Top line on first page
\AddToShipoutPictureBG*{%
  \AtPageUpperLeft{%
    \hspace*{13mm}\raisebox{-17mm}{%
      \textcolor{mycolor}{\rule{182mm}{1.2mm}}
    }%
  }%
}

% Top line on subsequent pages
\AddToShipoutPictureBG{%
  \AtPageUpperLeft{%
    \ifnum\value{page}>1
      \hspace*{13mm}\raisebox{-11mm}{%
        \textcolor{mycolor}{\rule{182mm}{0.3mm}}
      }%
    \fi
  }%
}

% Bottom line on all pages
\AddToShipoutPictureBG{%
  \AtPageLowerLeft{%
    \ifnum\value{page}>0
      \hspace*{13mm}\raisebox{12mm}{%
        \textcolor{mycolor}{\rule{182mm}{0.3mm}}
      }%
    \fi
  }%
}

}
